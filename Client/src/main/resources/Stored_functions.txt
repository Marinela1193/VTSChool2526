//passed subjects

CREATE OR REPLACE FUNCTION _da_vtschool_2526.subjectsPassed_mps_2526(student_idcard TEXT)
RETURNS TABLE(firstname TEXT, scoreDTO INT, subject_name TEXT) AS $$
SELECT s.firstname, sc.scoreDTO, sub.code
FROM _da_vtschool_2526.studentDTOS s
         JOIN _da_vtschool_2526.enrollments e ON e.studentDTO = s.idcard
         JOIN _da_vtschool_2526.scoreDTOS sc ON sc.enrollment_id = e.code
         JOIN _da_vtschool_2526.subjects sub ON sub.code = sc.subject_id
WHERE sc.scoreDTO <= 5 AND s.idcard = student_idcard;
$$ LANGUAGE sql;

//pending subjects

CREATE OR REPLACE FUNCTION _da_vtschool_2526.subjectsPending_mps_2526(student_idcard TEXT)
RETURNS TABLE(firstname TEXT, scoreDTO INT, subject_name TEXT) AS $$
SELECT s.firstname, sc.scoreDTO, sub.code
FROM _da_vtschool_2526.studentDTOS s
         JOIN _da_vtschool_2526.enrollments e ON e.studentDTO = s.idcard
         JOIN _da_vtschool_2526.scoreDTOS sc ON sc.enrollment_id = e.code
         JOIN _da_vtschool_2526.subjects sub ON sub.code = sc.subject_id
WHERE sc.scoreDTO < 5 AND s.idcard = student_idcard;
$$ LANGUAGE sql;
